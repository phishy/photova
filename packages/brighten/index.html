import { useEffect, useRef } from 'react';
import { EditorUI } from 'brighten';

export default function Editor({ src }) {
  const ref = useRef();
  
  useEffect(() => {
    const editor = new EditorUI({
      container: ref.current,
      image: src
    });
    return () => editor.destroy();
  }, [src]);

  return <div ref={ref} />;
}

'use client';
import { useEffect, useRef } from 'react';

export default function Editor({ src }) {
  const ref = useRef();

  useEffect(() => {
    import('brighten').then(({ EditorUI }) => {
      new EditorUI({
        container: ref.current,
        image: src
      });
    });
  }, [src]);

  return <div ref={ref} />;
}
